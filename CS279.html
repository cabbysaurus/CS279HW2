<form id="myForm">

  <select id="selectNumber">

    <option>Choose a number</option>

  </select>

</form>


<script>

var select = document.getElementById("selectNumber"); 

var options1 = ["A", "B", "C", "D"]; 

var options2 = ["E", "F", "G", "H"]; 

var options3 = ["I", "J", "K", "L"]; 

var options4 = ["M", "N", "O", "P"]; 

options1 = shuffle(options1);

options2 = shuffle(options2);

options3 = shuffle(options3);

options4 = shuffle(options4);


var options = options1.concat(options2.concat(options3.concat(options4)));


for(var i = 0; i < options.length; i++) {

    var opt = options[i];

    var el = document.createElement("option");

    el.textContent = opt;

    el.value = opt;

    select.appendChild(el);

}


function shuffle(array) {

  var currentIndex = array.length, temporaryValue, randomIndex;


  // While there remain elements to shuffle...

  while (0 !== currentIndex) {


    // Pick a remaining element...

    randomIndex = Math.floor(Math.random() * currentIndex);

    currentIndex -= 1;


    // And swap it with the current element.

    temporaryValue = array[currentIndex];

    array[currentIndex] = array[randomIndex];

    array[randomIndex] = temporaryValue;

  }


  return array;

}


</script>
